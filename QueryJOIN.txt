SELECT CONCAT(m.naam ,' ',m.voorn) as 'FULL NAAM'
		FROM medewerkers m 
	INNER JOIN afdelingen a 
	ON m.afd = a.anr 
	INNER JOIN inschrijvingen i
	ON m.mnr = i.cursist 
	INNER JOIN 
	uitvoeringe u
	ON
	CONCAT(i.cursus,i.begindatum) = CONCAT(u.cursus,ubegindatum)
	AND a.naam = u.locatie
	GO

SELECT CONCAT(m.naam ,' ' ,m.voorn),
		CONCAT(d.naam,' ' ,d.voorn) AS 'docent' ,
		c.omschrijving AS 'cursus'
		FROM inschrijvingen i
		INNER JOIN uitvoeringen u
		ON CONCAT(i.cursus,i.begindatum) =CONCAT(u.cursus,u.begindatum)
		INNER JOIN cursussen c
		ON u.cursus = c.code
		INNER JOIN medewerkers m 
		ON m.mnr = i.cursist 
		INNER JOIN medewerkers d
		ON d.mnr = u.docent
		ORDER BY d.naam
		GO

SELECT 
	COALESCE(m.naam,'==onbekend==') AS 'Docent',
	u.begindatum,c.omschrijving
	FROM medewerkers m 
	FULL OUTER JOIN uitvoeringen u
	ON 
	m.mnr = u.docent
	INNER JOIN cursussen c
	ON u.cursus = c.code
GO

SELECT 
	m.naam AS 'cursist'
	,b.naam AS 'chef'
	,a.naam AS  'Afdeling'
 	,d.naam  AS 'Docent'
	,c.omschrijving
	FROM inschrijvingen i
	INNER JOIN medewerkers m
	ON m.mnr = i.cursist 
	INNER JOIN medewerkers b
	ON m.chef = b.mnr
	INNER JOIN uitvoeringen u
	ON CONCAT(i.cursus,i.begindatum) =CONCAT(u.cursus,u.begindatum)
	FULL OUTER JOIN  medewerkers d 
	ON d.mnr = u.docent
	INNER JOIN afdelingen a
	on m.afd = a.anr 
	INNER JOIN cursussen c
	ON c.code = u.cursus
	GO
